2""" 

3Code Genetics pour Athalia/Arkalia 

4- Croisement, mutation, sélection, évolution de solutions IA 

5""" 

6from typing import List, Callable 

7import random 

8 

9class CodeGenetics: 

10 def crossover(self, solutions: List[str]) -> str: 

11 """ 

12 Croisement de solutions : mélange aléatoire de fragments de chaque solution. 

13 :param solutions: Liste de solutions (str) 

14 :return: Nouvelle solution croisée 

15 """ 

16 if not solutions: 

17 return "" 

18 fragments = [sol.split() for sol in solutions] 

19 length = min(len(f) for f in fragments) 

20 child = [] 

21 for i in range(length): 

22 parent = random.choice(fragments) 

23 child.append(parent[i]) 

24 return ' '.join(child) 

25 

26 def mutate(self, solution: str, mutation_rate: float = 0.1) -> str: 

27 """ 

28 Mutation simple : modifie aléatoirement des mots de la solution. 

29 :param solution: Solution à muter 

30 :param mutation_rate: Taux de mutation (0-1) 

31 :return: Solution mutée 

32 """ 

33 words = solution.split() 

34 for i in range(len(words)): 

35 if random.random() < mutation_rate: 

36 words[i] = f"MUT_{random.randint(0,999)}" 

37 return ' '.join(words) 

38 

39 def select(self, solutions: List[str], scorer: Callable[[str], float], top_k: int = 2) -> List[str]: 

40 """ 

41 Sélectionne les meilleures solutions selon un score. 

42 :param solutions: Liste de solutions 

43 :param scorer: Fonction de scoring (str -> float) 

44 :param top_k: Nombre de solutions à garder 

45 :return: Liste des meilleures solutions 

46 """ 

47 scored = sorted(solutions, key=scorer, reverse=True) 

48 return scored[:top_k] 

49 

50 def evolve(self, solutions: List[str], scorer: Callable[[str], float], generations: int = 3, mutation_rate: float = 0.1) -> str: 

51 """ 

52 Fait évoluer les solutions sur plusieurs générations (croisement, mutation, sélection). 

53 :param solutions: Liste initiale 

54 :param scorer: Fonction de scoring 

55 :param generations: Nombre de générations 

56 :param mutation_rate: Taux de mutation 

57 :return: Meilleure solution finale 

58 """ 

59 population = solutions[:] 

60 for _ in range(generations): 

61 # Croisement 

62 children = [self.crossover(random.sample(population, min(2, len(population)))) for _ in range(len(population))] 

63 # Mutation 

64 mutated = [self.mutate(child, mutation_rate) for child in children] 

65 # Sélection 

66 population = self.select(population + mutated, scorer, top_k=len(solutions)) 

67 return population[0] if population else ""

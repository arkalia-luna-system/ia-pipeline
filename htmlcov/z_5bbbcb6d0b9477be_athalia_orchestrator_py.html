386 def distill_ia_responses(self, prompt: str, models: Optional[List[str]] = None, strategy: str = 'voting') -> str: 

387 """ 

388 Interroge Qwen, Mistral, Mock (via RobustAI), distille les réponses et retourne la meilleure. 

389 """ 

390 ai = RobustAI() 

391 # Détection dynamique des modèles disponibles 

392 fallback_models = ai.fallback_chain 

393 if models: 

394 # Filtrer la chaîne de fallback selon la liste demandée 

395 fallback_models = [m for m in fallback_models if m.value in models or m.name in models] 

396 responses = [] 

397 for model in fallback_models: 

398 try: 

399 # Utilise le prompt blueprint pour l'exemple, peut être adapté 

400 res = ai._call_model(model, prompt) 

401 if res: 

402 responses.append(res) 

403 except Exception as e: 

404 responses.append(f"Erreur {model.value}: {e}") 

405 if not responses: 

406 return "[Aucune réponse IA disponible]" 

407 from athalia_core.distillation.response_distiller import distill_responses 

408 return distill_responses(responses, strategy=strategy)
